Step 4 – Replace localStorage with Supabase calls
Your CRM uses a global DataStore object that reads/writes to localStorage. We need to replace it with calls to Supabase.

Open data.js. We’ll modify the DataStore methods to use Supabase instead of localStorage. Because your project is large, we’ll focus on the most important methods: getAll, getById, create, update, delete. We’ll keep the same API so that other parts of your code don’t break.

Important: This change will make your app live‑online only – it will no longer work offline. We’ll also need to adapt your authentication (the Auth module) to use Supabase Auth.

4.1 Modify data.js – replace storage functions
Open data.js and replace the whole file with this new version. I’ll show you the key changes; you can copy the entire new file from the end of this guide.

javascript
// data.js – Supabase version
const DataStore = (() => {
    const TABLES = [ ... ]; // same as before

    // We'll use the global supabase client
    const supabaseClient = window.supabase;

    const getAll = async (tableName) => {
        const { data, error } = await supabaseClient
            .from(tableName)
            .select('*');
        if (error) throw error;
        return data || [];
    };

    const getById = async (tableName, id) => {
        const { data, error } = await supabaseClient
            .from(tableName)
            .select('*')
            .eq('id', id)
            .single();
        if (error && error.code !== 'PGRST116') throw error;
        return data || null;
    };

    const create = async (tableName, data) => {
        const newRecord = { ...data, created_at: new Date().toISOString() };
        const { data: inserted, error } = await supabaseClient
            .from(tableName)
            .insert(newRecord)
            .select()
            .single();
        if (error) throw error;
        // Dispatch event
        window.dispatchEvent(new CustomEvent('dataChanged', {
            detail: { table: tableName, action: 'create', id: inserted.id }
        }));
        return inserted;
    };

    const update = async (tableName, id, data) => {
        const { data: updated, error } = await supabaseClient
            .from(tableName)
            .update(data)
            .eq('id', id)
            .select()
            .single();
        if (error) throw error;
        window.dispatchEvent(new CustomEvent('dataChanged', {
            detail: { table: tableName, action: 'update', id }
        }));
        return updated;
    };

    const deleteRecord = async (tableName, id) => {
        const { error } = await supabaseClient
            .from(tableName)
            .delete()
            .eq('id', id);
        if (error) throw error;
        window.dispatchEvent(new CustomEvent('dataChanged', {
            detail: { table: tableName, action: 'delete', id }
        }));
        return true;
    };

    const query = async (tableName, filters) => {
        let query = supabaseClient.from(tableName).select('*');
        for (const [key, value] of Object.entries(filters)) {
            query = query.eq(key, value);
        }
        const { data, error } = await query;
        if (error) throw error;
        return data || [];
    };

    // The rest of the DataStore methods (init, on, emit) need to be adapted to work asynchronously.
    // For simplicity, we'll make all functions async and keep event listeners.

    return {
        getAll,
        getById,
        create,
        update,
        delete: deleteRecord,
        query,
        on: (event, cb) => { ... },  // same as before
        emit: (event, data) => { ... },
        init: () => { ... } // maybe check tables exist, etc.
    };
})();
Note: Because we changed methods to be async, every place in your code that calls DataStore.getAll() will now need to use await. This is a big change. You must go through your script.js and add await before every DataStore call. For example:

DataStore.getAll('users') becomes await DataStore.getAll('users')

DataStore.create('prospects', data) becomes await DataStore.create('prospects', data)

Since you have many functions, you’ll need to systematically add async to any function that uses await. This is a large but necessary step.

4.2 Update your auth.js to use Supabase Auth
Supabase has its own authentication system. You’ll need to replace your simple localStorage‑based login with Supabase Auth. In auth.js:

javascript
const Auth = (() => {
    const supabase = window.supabase;

    const getCurrentUser = async () => {
        const { data: { user } } = await supabase.auth.getUser();
        return user;
    };

    const login = async (email, password) => {
        const { data, error } = await supabase.auth.signInWithPassword({ email, password });
        if (error) throw error;
        return data.user;
    };

    const logout = async () => {
        await supabase.auth.signOut();
    };

    return { getCurrentUser, login, logout };
})();
You’ll also need to change your login form to collect email/password instead of selecting a user from a dropdown. You can keep a dropdown for demo users, but you’d need to create those users in Supabase Auth first.

Write a migration script
You’ll need to insert each record into the corresponding Supabase table. Because you have many records, it’s easier to write a small Node.js script that uses the Supabase client to upload them. I’ll give you a simple example (you’ll need Node.js installed).

Create a new file migrate.js outside your project folder. Install dependencies:

bash
npm init -y
npm install @supabase/supabase-js
Then write:

javascript
const { createClient } = require('@supabase/supabase-js');
const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY);
const data = require('./exported-data.json'); // your saved JSON

const tables = {
    'fs_crm_users': 'users',
    'fs_crm_prospects': 'prospects',
    // ... map all table names
};

async function migrate() {
    for (const [localKey, tableName] of Object.entries(tables)) {
        const records = data[localKey];
        if (!records) continue;
        for (const record of records) {
            // Remove any fields that don't exist in Supabase schema
            const { error } = await supabase.from(tableName).insert(record);
            if (error) console.error(`Error inserting into ${tableName}:`, error);
            else console.log(`Inserted into ${tableName}`);
        }
    }
}

migrate();
Run it with:

bash
SUPABASE_URL=your_url SUPABASE_KEY=your_key node migrate.js
(Use the service role key from Supabase for this, not the anon key.)

This is a simplified approach – you may need to handle conflicts, set proper IDs, etc.

        'audit_logs', 'security_incidents', 'consent_records', 'consent_revocations',
        'dsar_requests', 'dsar_reports', 'retention_jobs', 'archived_records',
        'encryption_keys', 'mfa_devices', 'login_attempts', 'ip_whitelist', 'data_classification',
        'tenants', 'tenant_settings', 'health_checks', 'system_alerts', 'performance_metrics',
        'performance_warnings', 'backups', 'backup_schedules', 'deployments', 'system_config',
        'system_logs', 'maintenance_windows', 'audit_logs_archive',
        'event_attendees', 'agent_event_attendees', 'case_studies',
        'promotion_packages',
        'products', 'promotions',
        'appointment_locations'
    ];

    const _listeners = {};

    const supabaseClient = window.supabase;

    const getAll = async (tableName) => {
        if (!TABLES.includes(tableName)) return [];
        const { data, error } = await supabaseClient.from(tableName).select('*');
        if (error) throw error;
        return data || [];
    };

    const getById = async (tableName, id) => {
        const { data, error } = await supabaseClient
            .from(tableName)
            .select('*')
            .eq('id', id)
            .single();
        if (error && error.code !== 'PGRST116') throw error;
        return data || null;
    };

    const create = async (tableName, data) => {
        const newRecord = { ...data, created_at: new Date().toISOString() };
        const { data: inserted, error } = await supabaseClient
            .from(tableName)
            .insert(newRecord)
            .select()
            .single();
        if (error) throw error;
        DataStore.emit(`${tableName}:created`, inserted);
        window.dispatchEvent(new CustomEvent('dataChanged', {
            detail: { table: tableName, action: 'create', id: inserted.id }
        }));
        return inserted;
    };

    const update = async (tableName, id, data) => {
        const { data: updated, error } = await supabaseClient
            .from(tableName)
            .update(data)
            .eq('id', id)
            .select()
            .single();
        if (error) throw error;
        DataStore.emit(`${tableName}:updated`, updated);
        window.dispatchEvent(new CustomEvent('dataChanged', {
            detail: { table: tableName, action: 'update', id: id }
        }));
        return updated;
    };

    const deleteRecord = async (tableName, id) => {
        const { error } = await supabaseClient
            .from(tableName)
            .delete()
            .eq('id', id);
        if (error) throw error;
        DataStore.emit(`${tableName}:deleted`, id);
        window.dispatchEvent(new CustomEvent('dataChanged', {
            detail: { table: tableName, action: 'delete', id: id }
        }));
        return true;
    };

    const query = async (tableName, filters) => {
        let query = supabaseClient.from(tableName).select('*');
        for (const [key, value] of Object.entries(filters)) {
            query = query.eq(key, value);
        }
        const { data, error } = await query;
        if (error) throw error;
        return data || [];
    };

    const on = (eventName, callback) => {
        if (!_listeners[eventName]) _listeners[eventName] = [];
        _listeners[eventName].push(callback);
    };

    const emit = (eventName, data) => {
        if (!_listeners[eventName]) return;
        _listeners[eventName].forEach(cb => cb(data));
    };

    const init = async () => {
        // Optionally check if tables exist
        console.log('DataStore initialized with Supabase');
    };

    return {
        getAll,
        getById,
        create,
        update,
        delete: deleteRecord,
        query,
        on,
        emit,
        init
    };
})();
